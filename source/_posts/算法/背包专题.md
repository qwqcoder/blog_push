# 多重背包

==分别使用01和多重背包==

```c++
const int N = 50000;

int dp[N];
int n, m;
int v[N], w[N], s[N];

signed main() {
    cin >> n >> m;
    for (int i = 1; i <= n; i++)
    {
        cin >> v[i] >> w[i] >> s[i];
        //如果所有当前项的花费超过总的背包体积，约等于当前项无限多，完全背包处理
        if (s[i] * v[i] >= m) {
            for (int j = v[i]; j <= m; j++)
                dp[j] = max(dp[j], dp[j - v[i]] + w[i]);
        }
        else
        {
            //二进制分解使用01背包处理
            int k = 1, c = s[i];
            while (k <= c)
            {
                for (int j = m; j >= k * v[i]; j--)
                    dp[j] = max(dp[j], dp[j - k * v[i]] + w[i] * k);
                c -= k;
                k <<= 1;
            }
            if (c)
                for (int j = m; j >= c * v[i]; j--)
                    dp[j] = max(dp[j], dp[j - c * v[i]] + c * w[i]);
        }
    }
    cout << dp[m];
    return 0;
}
```

# 混合背包

```c++
#include <bits/stdc++.h>
using namespace std;
int dp[1100];
int q[1100];
int hh = -1, tt = 0;
signed main(){
    int n, m;cin >> n >> m;
    for(int i = 1;i <= n; i++)
    {
        int v, w, s;
        cin >> v >> w >> s;
        if(s < 0){
            for(int j = m;j >= v;j --)
                dp[j] = max(dp[j - v] + w, dp[j]);
        }
        else if(s == 0)
        {
            for(int j = v;j <= m;j ++)
                dp[j] = max(dp[j - v] + w, dp[j]);
        }
        else{
            int k = 1, c = s;
            while (k <= c)
            {
                for (int j = m; j >= k * v; j--)
                    dp[j] = max(dp[j], dp[j - k * v] + w * k);
                c -= k;
                k <<= 1;
            }
            if(c)
            for (int j = m; j >= c * v; j--)
                dp[j] = max(dp[j], dp[j - c * v] + c * w);
        }
    }
    cout << dp[m];
    return 0;
}
```
